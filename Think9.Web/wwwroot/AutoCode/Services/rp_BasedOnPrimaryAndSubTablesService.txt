        /// <summary>
        /// 处理Value与Text转化
        /// </summary>
        /// <param name="SelectList">Value与Text数据列表</param>
        /// <param name="list">数据列表</param>
        /// <param name="user">当前用户</param>
        private void TransformList(IEnumerable<valueTextEntity> SelectList, List<dynamic> list, CurrentUserEntity user = null)
        {
            foreach (BasedOnPrimaryAndSubTablesModel obj in list)
            {

				//供应商
				foreach (valueTextEntity sel in SelectList.Where(x => x.ClassID == "inSupplier"))
				{
					if (obj.inSupplier == sel.Value)
					{
						obj.inSupplier = sel.Text;
						break;
					}
				}
				//审核
				foreach (valueTextEntity sel in SelectList.Where(x => x.ClassID == "inAuditing"))
				{
					if (obj.inAuditing == sel.Value)
					{
						obj.inAuditing = sel.Text;
						break;
					}
				}
				//单位
				foreach (valueTextEntity sel in SelectList.Where(x => x.ClassID == "v4"))
				{
					if (obj.v4 == sel.Value)
					{
						obj.v4 = sel.Text;
						break;
					}
				}
            }
        }

		/// <summary>
		/// 获取下拉选择、多选、单选数据源
		/// 处理列表显示时value与text转化，动态数据源来源于为指标指定的数据规范
		/// </summary>
		/// <param name="from">add、edit或list，list表明从列表页面查询条件跳转而来</param>
		/// <param name="param">条件参数</param>
		/// <returns></returns>
		public IEnumerable<valueTextEntity> GetSelectList(string from, object param)
		{
			DataTable dt = DataTableHelp.NewValueTextDt();

			//审核
			sql = "SELECT DetailCode AS id,DetailName AS name FROM sys_itemsdetail WHERE ItemCode = 'SHZT' ORDER BY OrderNo";
			foreach (DataRow dr in comService.GetDataTable(sql, param).Rows)
			{
				DataRow row = dt.NewRow();
				row["ClassID"] = "inAuditing";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}
			//供应商
			sql = "SELECT inCoding AS id,inName AS name FROM tb_Supplier ORDER BY state";
			foreach (DataRow dr in comService.GetDataTable(sql, param).Rows)
			{
				DataRow row = dt.NewRow();
				row["ClassID"] = "inSupplier";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}
			//编码
			sql = "SELECT inCoding AS id,inName AS name FROM tb_ProductManagement ORDER BY state";
			foreach (DataRow dr in comService.GetDataTable(sql, param).Rows)
			{
				DataRow row = dt.NewRow();
				row["ClassID"] = "v1";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}
			//单位
			sql = "SELECT inCoding AS id,inName AS name FROM tb_JLDW ORDER BY state";
			foreach (DataRow dr in comService.GetDataTable(sql, param).Rows)
			{
				DataRow row = dt.NewRow();
				row["ClassID"] = "v4";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}

			return DataTableHelp.ToEnumerable<valueTextEntity>(dt);//DataTable转换为IEnumerable
		}

        /// <summary>
        /// 获取查询字符串
        /// </summary>
        /// <param name="list">查询条件list</param>
        /// <returns>返回查询字符串</returns>
        private string GetSearchWhere(IEnumerable<ControlEntity> list)
		{
			string where = " WHERE (1 = 1)";
			
			//审核
			str = BasicHelp.GetValueFrmList(list, "inAuditing").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementOfGoods.inAuditing = @inAuditing)"; 
			}
			
			//采购日期
			str = BasicHelp.GetValueFrmList(list, "inPurchaseDate").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementOfGoods.inPurchaseDate >= @inPurchaseDate)"; 
			}
			str = BasicHelp.GetValueFrmList(list, "inPurchaseDate_Exa").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementOfGoods.inPurchaseDate <= @inPurchaseDate_Exa)"; 
			}
			
			//供应商
			str = BasicHelp.GetValueFrmList(list, "inSupplier").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementOfGoods.inSupplier = @inSupplier)"; 
			}
			
			//名称
			str = BasicHelp.GetValueFrmList(list, "v2").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementDetails.v2 LIKE  @v2)"; 
			}
			
			//规格型号
			str = BasicHelp.GetValueFrmList(list, "v3").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementDetails.v3 LIKE  @v3)"; 
			}
			
			//数量
			str = BasicHelp.GetValueFrmList(list, "v6").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementDetails.v6 >= @v6)"; 
			}
			str = BasicHelp.GetValueFrmList(list, "v6_Exa").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementDetails.v6 <= @v6_Exa)"; 
			}
			
			//金额合计
			str = BasicHelp.GetValueFrmList(list, "inTotalAmount").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementOfGoods.inTotalAmount >= @inTotalAmount)"; 
			}
			str = BasicHelp.GetValueFrmList(list, "inTotalAmount_Exa").Trim();
			if (!string.IsNullOrEmpty(str))
			{
				where += " AND (tb_ProcurementOfGoods.inTotalAmount <= @inTotalAmount_Exa)"; 
			}
			return where;
		}

        /// 查询条件list转化为Model

        /// <returns></returns>
        private BasedOnPrimaryAndSubTablesModel GetParamSearch(IEnumerable<ControlEntity> list)
		{
			
			string _inAuditing = BasicHelp.GetValueFrmList(list, "inAuditing").Trim();//审核
			string _inPurchaseDate = BasicHelp.GetValueFrmList(list, "inPurchaseDate").Trim();//采购日期
			string _inPurchaseDate_Exa = BasicHelp.GetValueFrmList(list, "inPurchaseDate_Exa").Trim();//采购日期
			string _inSupplier = BasicHelp.GetValueFrmList(list, "inSupplier").Trim();//供应商
			string _v2 = BasicHelp.GetValueFrmList(list, "v2").Trim();//名称
			string _v3 = BasicHelp.GetValueFrmList(list, "v3").Trim();//规格型号
			string _v6 = BasicHelp.GetValueFrmList(list, "v6").Trim();//数量
			string _v6_Exa = BasicHelp.GetValueFrmList(list, "v6_Exa").Trim();//数量
			string _inTotalAmount = BasicHelp.GetValueFrmList(list, "inTotalAmount").Trim();//金额合计
			string _inTotalAmount_Exa = BasicHelp.GetValueFrmList(list, "inTotalAmount_Exa").Trim();//金额合计
			return new BasedOnPrimaryAndSubTablesModel{ inAuditing = _inAuditing, inPurchaseDate = ExtConvert.ToDateOrNull(_inPurchaseDate), inPurchaseDate_Exa = ExtConvert.ToDateOrNull(_inPurchaseDate_Exa), inSupplier = _inSupplier, v2 = string.Format("%{0}%", _v2), v3 = string.Format("%{0}%", _v3), v6 = ExtConvert.ToDecimalOrNull(_v6), v6_Exa = ExtConvert.ToDecimalOrNull(_v6_Exa), inTotalAmount = ExtConvert.ToDecimalOrNull(_inTotalAmount), inTotalAmount_Exa = ExtConvert.ToDecimalOrNull(_inTotalAmount_Exa) };
		}

        /// <summary>
        /// 获取对象并向查询条件赋初始值
        /// </summary>
        /// <param name="request">request不为空时（通常为从其他页面打开）</param>
        /// <returns></returns>
        public BasedOnPrimaryAndSubTablesModel GetModel(Microsoft.AspNetCore.Http.HttpRequest request)
        {
            BasedOnPrimaryAndSubTablesModel model = new BasedOnPrimaryAndSubTablesModel();
            if (request != null)
            {
				if (!string.IsNullOrEmpty(request.Query["inAuditing"]))
				{
					model.inAuditing = request.Query["inAuditing"].ToString();//审核
				}
				if (!string.IsNullOrEmpty(request.Query["inAuditing_Exa"]))
				{
					model.inAuditing_Exa = request.Query["inAuditing_Exa"].ToString();//审核
				}
				if (!string.IsNullOrEmpty(request.Query["inSupplier"]))
				{
					model.inSupplier = request.Query["inSupplier"].ToString();//供应商
				}
				if (!string.IsNullOrEmpty(request.Query["inSupplier_Exa"]))
				{
					model.inSupplier_Exa = request.Query["inSupplier_Exa"].ToString();//供应商
				}
				if (!string.IsNullOrEmpty(request.Query["inTotalAmount"]) && ValidatorHelper.IsNumberic(request.Query["inTotalAmount"].ToString()))
				{
					model.inTotalAmount = ExtConvert.ToDecimalOrNull(request.Query["inTotalAmount"].ToString());//金额合计
				}
				if (!string.IsNullOrEmpty(request.Query["inTotalAmount_Exa"]) && ValidatorHelper.IsNumberic(request.Query["inTotalAmount_Exa"].ToString()))
				{
					model.inTotalAmount_Exa = ExtConvert.ToDecimalOrNull(request.Query["inTotalAmount_Exa"].ToString());//金额合计
				}
				if (!string.IsNullOrEmpty(request.Query["inPurchaseDate"]) && ValidatorHelper.IsDateTime(request.Query["inPurchaseDate"].ToString()))
				{
					model.inPurchaseDate = ExtConvert.ToDateOrNull(request.Query["inPurchaseDate"].ToString());//采购日期
				}
				if (!string.IsNullOrEmpty(request.Query["inPurchaseDate_Exa"]) && ValidatorHelper.IsDateTime(request.Query["inPurchaseDate_Exa"].ToString()))
				{
					model.inPurchaseDate_Exa = ExtConvert.ToDateOrNull(request.Query["inPurchaseDate_Exa"].ToString());//采购日期
				}
				if (!string.IsNullOrEmpty(request.Query["v2"]))
				{
					model.v2 = request.Query["v2"].ToString();//名称
				}
				if (!string.IsNullOrEmpty(request.Query["v3"]))
				{
					model.v3 = request.Query["v3"].ToString();//规格型号
				}
				if (!string.IsNullOrEmpty(request.Query["v6"]) && ValidatorHelper.IsNumberic(request.Query["v6"].ToString()))
				{
					model.v6 = ExtConvert.ToDecimalOrNull(request.Query["v6"].ToString());//数量
				}
				if (!string.IsNullOrEmpty(request.Query["v6_Exa"]) && ValidatorHelper.IsNumberic(request.Query["v6_Exa"].ToString()))
				{
					model.v6_Exa = ExtConvert.ToDecimalOrNull(request.Query["v6_Exa"].ToString());//数量
				}
            }
            return model;
        }
