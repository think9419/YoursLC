        {
            string err = "";
            int num = gridlist.Count();
            if (num < 1)
            {
				return "";
            }
            string updateFields = GetUpdateFields();//可修改字段，排除了隐藏列
            if (!string.IsNullOrEmpty(updateFields))
            {
			    //唯一及主键检测
                DataTable dtIndex = TbIndex.GetBKAndUniqueDataTable(_gridtbid);
                if (dtIndex.Rows.Count > 0)
                {
                    foreach (Subtable06Model obj in gridlist)
                    {
                        err = Think9.Services.Com.CheckCom.CheckGridTbValueBKAndUnique(listid, obj.Id, _gridtbid, obj, DataTableHelp.ModelToDataTable<Subtable06Model>(obj), dtIndex);
                        if (err != "")
                        {
                            return err;
                        }
                    }
                }
                foreach (Subtable06Model obj in gridlist)
                {

                    if (base.UpdateById(obj, updateFields))
                    {
                        if (string.IsNullOrEmpty(isFinish))
                        {
                            List<string> resultList = AfterGridEdit(ref err, listid.ToString(), prcno, user, model, obj, obj.Id.ToString());
                            Record.AddResultList("system", listid.ToString(), _flowid, resultList, "#自定义事件AfterGridEdit#");
                        }
                        else
                        {
                            List<string> resultList = AfterGridFinish(ref err, listid.ToString(), prcno, user, model, obj, obj.Id.ToString());
                            Record.AddResultList("system", listid.ToString(), _flowid, resultList, "#自定义事件AfterGridFinish#");
                        }
                    }
                }
				Record.Add(user == null ? "!NullEx" : user.Account, listid.ToString(), _flowid, "#子表编辑数据##" + _gridtbid + "##N{" + num + "}#");
            }
            return err;
        }

	   /// <summary>
	   /// 将前端返回的list转化为子表对象，并校验
	   /// </summary>
	   /// <param name="strErr">错误信息</param>
	   /// <param name="listid">主表主键</param>

	   /// <param name="flag">标志位，不为空时表示增加一条数据</param>
	   /// <returns></returns>
	   public static List<Subtable06Model> CheckedAndGetList(ref string strErr, long listid, IEnumerable<GridListEntity> list, string flag)
		{
			string id = "";
			string row = "";
			string err = "";
			string temp = "";
			bool isAdd = false;

			List<Subtable06Model> mlist = new List<Subtable06Model>();
			foreach (GridListEntity obj in list)
			{
				isAdd = false;
				if (!string.IsNullOrEmpty(flag))//添加一条数据
				{
					if (obj.flag.StartsWith(flag))
					{
						isAdd = true;
					}
				}
				else//编辑列表数据
				{
                    //排除首行即增加行
                    if (obj.flag.StartsWith("#" + _gridtbid + "#") && !obj.flag.EndsWith("#0#0#"))
                    {
                        isAdd = true;
                    }
				}

				if (isAdd)
				{
					err = "";
					BasicHelp.GetRowAndIdByFlag(obj.flag, ref id, ref row);

					//弹出选择
					if (string.IsNullOrEmpty(obj.v1))
					{
						err += row + "弹出选择是必填项";
					}

					if (string.IsNullOrEmpty(err))
					{
						Subtable06Model model = new Subtable06Model();
						model.TbId = _gridtbid;
						model.ListId = listid;
						model.Id = long.Parse(id);
						model.v1 = obj.v1 == null ? "" : obj.v1;
						model.v2 = obj.v2 == null ? "" : obj.v2;
						model.v3 = obj.v3 == null ? "" : obj.v3;
						model.v4 = obj.v4 == null ? "" : obj.v4;
						model.v5 = obj.v5 == null ? "" : obj.v5;
						model.v6 = obj.v6 == null ? "" : obj.v6;
						model.v7 = obj.v7 == null ? "" : obj.v7;

						mlist.Add(model);
					}
					else
					{
						strErr += _gridtbname + ":" + err + "<br>";
					}
				}
			}
			return mlist;
		}

		public IEnumerable<valueTextEntity> GetSelectList()
		{
			DataTable dt = DataTableHelp.NewValueTextDt();

			//下拉选择
			sql = "SELECT inCoding AS id,inName AS name FROM tb_JLDW ORDER BY state";
			foreach (DataRow dr in comService.GetDataTable(sql).Rows)
			{
				this.v2 += "<option value='"+ dr["id"].ToString() +"'>" + dr["name"].ToString() + "</option>";
				DataRow row = dt.NewRow();
				row["ClassID"] = "v2";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}
			//单位
			sql = "SELECT inCoding AS id,inName AS name FROM tb_JLDW ORDER BY state";
			foreach (DataRow dr in comService.GetDataTable(sql).Rows)
			{
				this.v5 += "<option value='"+ dr["id"].ToString() +"'>" + dr["name"].ToString() + "</option>";
				DataRow row = dt.NewRow();
				row["ClassID"] = "v5";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}
			//类别
			sql = "SELECT inCoding AS id,inName AS name FROM tb_HPLB ORDER BY state";
			foreach (DataRow dr in comService.GetDataTable(sql).Rows)
			{
				this.v6 += "<option value='"+ dr["id"].ToString() +"'>" + dr["name"].ToString() + "</option>";
				DataRow row = dt.NewRow();
				row["ClassID"] = "v6";
				row["Value"] = dr["id"].ToString();
				row["Text"] = dr["name"].ToString();
				dt.Rows.Add(row);
			}
            return DataTableHelp.ToEnumerable<valueTextEntity>(dt);//DataTable转换为IEnumerable
		}

		private object GetObj(string id, string listid, CurrentUserEntity user, ref ReadSingleValueModel model, ref Subtable06Model gridObj)
        {
            if (model == null)
            {
                ReadSingleValueService tbMain = new ReadSingleValueService();
                model = tbMain.GetByWhereFirst("WHERE listid=" + listid);
				model = model == null ? new ReadSingleValueModel() : model;//预防空对象
            }
            if (string.IsNullOrEmpty(model.createUser))
            {
                MainTBEntity entity = MainTBService.GetMainTBEntity(listid, _flowid);
                model.isLock = entity.isLock;
                model.createTime = entity.createTime;
                model.createUser = entity.createUser;

		/// <param name="grid">子表id</param>
		/// <param name="user">当前用户</param>
		/// <param name="controlslist">主表控件列表，可读取控件值</param>
		/// <param name="gridlist">子表控件列表，可读取控件值</param>

		/// <param name="value">触发数据读取的控件value值</param>
		/// <returns>返回List前端再解析</returns>
		public List<ControlEntity> ReadValue(ref string err, ref string listid, ref string grid, CurrentUserEntity user, IEnumerable<ControlEntity> controlslist, IEnumerable<ControlEntity> gridlist, string indexid, string value)
		{
			string tbname = "";
			string some = "";
			string where = "";
			string order = "";
			DataTable dt = new DataTable();
			List<ControlEntity> list = new List<ControlEntity>();
			
			if (indexid == "Subtable06v1")
			{
				//读取单值 读取货品信息
				tbname = "tb_ProductManagement";
				some = "tb_ProductManagement.inCategory AS v6,tb_ProductManagement.inName AS v3,tb_ProductManagement.inUnit AS v5,tb_ProductManagement.inSpecificationAndModel AS v4,tb_ProductManagement.inUnitPrice AS v7,tb_ProductManagement.inUnit AS v2";
				where = "WHERE tb_ProductManagement.inCoding = @v1";
				order = "ORDER BY listid";
				
				dt = comService.GetDataTable(tbname, some, where, order, new { v1 = value });
				if (dt.Rows.Count > 0)
				{
					//类别
					list.Add(new ControlEntity { ControlID = "v6", ControlValue = dt.Rows[0]["v6"].ToString(), ControlType = "2" });
					//名称
					list.Add(new ControlEntity { ControlID = "v3", ControlValue = dt.Rows[0]["v3"].ToString(), ControlType = "1" });
					//单位
					list.Add(new ControlEntity { ControlID = "v5", ControlValue = dt.Rows[0]["v5"].ToString(), ControlType = "2" });
					//规格型号
					list.Add(new ControlEntity { ControlID = "v4", ControlValue = dt.Rows[0]["v4"].ToString(), ControlType = "1" });
					//备注
					list.Add(new ControlEntity { ControlID = "v7", ControlValue = dt.Rows[0]["v7"].ToString(), ControlType = "1" });
					//下拉选择
					list.Add(new ControlEntity { ControlID = "v2", ControlValue = dt.Rows[0]["v2"].ToString(), ControlType = "2" });
				}
			}
			if (indexid == "Subtable06v2")
			{
				//读取单值 读取计量单位
				tbname = "tb_JLDW";
				some = "tb_JLDW.inName AS v3,tb_JLDW.inName AS v7,tb_JLDW.inCoding AS v5,tb_JLDW.inCoding AS v4,tb_JLDW.inName AS v1";
				where = "WHERE tb_JLDW.inCoding = @v2";
				order = "ORDER BY listid";
				
				dt = comService.GetDataTable(tbname, some, where, order, new { v2 = value });
				if (dt.Rows.Count > 0)
				{
					//名称
					list.Add(new ControlEntity { ControlID = "v3", ControlValue = dt.Rows[0]["v3"].ToString(), ControlType = "1" });
					//备注
					list.Add(new ControlEntity { ControlID = "v7", ControlValue = dt.Rows[0]["v7"].ToString(), ControlType = "1" });
					//单位
					list.Add(new ControlEntity { ControlID = "v5", ControlValue = dt.Rows[0]["v5"].ToString(), ControlType = "2" });
					//规格型号
					list.Add(new ControlEntity { ControlID = "v4", ControlValue = dt.Rows[0]["v4"].ToString(), ControlType = "1" });
					//弹出选择
					list.Add(new ControlEntity { ControlID = "v1", ControlValue = dt.Rows[0]["v1"].ToString(), ControlType = "1" });
				}
			}
			return list;
		}
